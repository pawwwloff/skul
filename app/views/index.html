Возварщает список всех авторов


Тип запроса - get

Пример - /api/author
Возвращаемые поля:
id - id автора;
name - имя автора;
email - емаил автора;
[
{
"id":1,
"name":"name",
"email":"email@email.com"
},
{
...
}
]

Возварщает список новостей автора по id


Тип запроса - get

Пример - /api/author/2/news
Возвращаемые поля:
id - id новости;
name - название новости;
text - описание новости;
category - id категории;
author - id автора;
categories - массив категорий состоит из id, name, parent;
[
{
"id":1,
"name":"name",
"text":"...",
"category":2,
"author":3,
"categories":[
{
"id":3,
"name":"repellendus",
"parent":2
},
{
...
}
]
},
{
...
}
]
